void SelectionSort(int a[] ,int size)

{

int i,j ,temp;


for(i=0;i<size-1;i++)


{
   
for(j=i+1;j<size;j++)

    {
        
if(a[i]<a[j])

         {
              

temp=a[i];

              a[i]=a[j];

              a[j]=temp;
          }
  
 

 

}
}
}
