void InsertionSort(int a[] ,int size)

{

int i,j ,temp;


for(i=1;i<size;i++)


{
   
for(j=0;j<i-1;j++)

    {
        
if(a[i]<a[j])

         {
              

temp=a[i];

              a[i]=a[j];

              a[j]=temp;
          }
  
 

 

}
}
}
